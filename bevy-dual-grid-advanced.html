#[derive(Clone, Copy, PartialEq, Eq, Debug)]
enum Terrain { Water = 0, Lava = 1, Dirt = 2, Grass = 3, Stone = 4, Concrete = 5 }
impl Terrain { fn priority(self) -> usize { self as usize } }

#[derive(Clone, Copy)]
struct Cell { terrain: Terrain, height: u8 }

#[derive(Resource)]
struct WorldGrid { w: i32, h: i32, cells: Vec<Cell> }

impl WorldGrid {
    fn new(w: i32, h: i32) -> Self {
        Self { w, h, cells: vec![Cell { terrain: Terrain::Dirt, height: 0 }; (w * h) as usize] }
    }
    fn cell(&self, x: i32, y: i32) -> Cell {
        if x < 0 || y < 0 || x >= self.w || y >= self.h {
            Cell { terrain: Terrain::Dirt, height: 0 }
        } else { self.cells[(y * self.w + x) as usize] }
    }
    fn priority(&self, x: i32, y: i32) -> usize { self.cell(x, y).terrain.priority() }
    fn height(&self, x: i32, y: i32) -> u8 { self.cell(x, y).height }
    fn set_terrain(&mut self, x: i32, y: i32, t: Terrain) -> bool {
        if x < 0 || y < 0 || x >= self.w || y >= self.h { return false; }
        let i = (y * self.w + x) as usize;
        if self.cells[i].terrain == t { return false; }
        self.cells[i].terrain = t; true
    }
    fn set_height(&mut self, x: i32, y: i32, hgt: u8) -> bool {
        if x < 0 || y < 0 || x >= self.w || y >= self.h { return false; }
        let i = (y * self.w + x) as usize;
        if self.cells[i].height == hgt { return false; }
        self.cells[i].height = hgt; true
    }
}

fn layer_mask(grid: &WorldGrid, layer: usize, dx: i32, dy: i32) -> usize {
    let tl = (grid.priority(dx - 1, dy)     >= layer) as usize;
    let tr = (grid.priority(dx, dy)         >= layer) as usize;
    let bl = (grid.priority(dx - 1, dy - 1) >= layer) as usize;
    let br = (grid.priority(dx, dy - 1)     >= layer) as usize;
    (tl << 3) | (tr << 2) | (bl << 1) | br
}
fn has_south_cliff(grid: &WorldGrid, x: i32, y: i32) -> bool {
    grid.height(x, y) > grid.height(x, y - 1)
}
fn rebuild_walls(grid: &WorldGrid, walls: &mut Walls) {
    walls.0.clear();

    }}
}

fn paint_cells(
    buttons: Res<ButtonInput<MouseButton>>,
    brush: Res<Brush>,
    window: Single<&Window, With<PrimaryWindow>>,
    camera: Single<(&Camera, &GlobalTransform)>,
    mut grid: ResMut<WorldGrid>,
    mut changed: MessageWriter<CellChanged>,
) {
    let left = buttons.pressed(MouseButton::Left);
    let right = buttons.pressed(MouseButton::Right);
    if !left && !right { return; }

    let Some(cursor) = window.cursor_position() else { return; };
    let (camera, cam_transform) = *camera;
    let Ok(world_pos) = camera.viewport_to_world_2d(cam_transform, cursor) else { return; };
    let x = (world_pos.x / TILE).round() as i32;
    let y = (world_pos.y / TILE).round() as i32;

    let did_change = match *brush {
        Brush::Terrain(t) => grid.set_terrain(x, y, if left { t } else { Terrain::Dirt }),
        Brush::Elevation => grid.set_height(x, y, if left { 1 } else { 0 }),
    };
    if did_change { changed.write(CellChanged { x, y }); }
}

fn refresh_display(
    mut changed: MessageReader<CellChanged>,
    grid: Res<WorldGrid>,
    mut commands: Commands,
    mut walls: ResMut<Walls>,
    cliff_tiles: Res<CliffTiles>,
    mut tiles: Query<(&DisplayTile, &Layer, &mut Sprite)>,
    cliffs: Query<Entity, With<Cliff>>,
) {
    let dirty: Vec<(i32, i32)> = changed.read()
        .flat_map(|m| [(m.x, m.y), (m.x + 1, m.y), (m.x, m.y + 1), (m.x + 1, m.y + 1)])
        .collect();
    if dirty.is_empty() { return; }

    for (tile, layer, mut sprite) in &mut tiles {
        if dirty.contains(&(tile.dx, tile.dy)) {
            if let Some(atlas) = sprite.texture_atlas.as_mut() {
                atlas.index = layer.0 * 16 + layer_mask(&grid, layer.0, tile.dx, tile.dy);
            }
        }
    }
    rebuild_walls(&grid, &mut walls);
    for e in &cliffs { commands.entity(e).despawn(); }
    respawn_cliffs(&mut commands, &grid, &cliff_tiles);
}

fn cycle_weather(keys: Res<ButtonInput<KeyCode>>, mut weather: ResMut<Weather>) {
    if keys.just_pressed(KeyCode::KeyW) { *weather = weather.next(); }
}

fn apply_weather(
    weather: Res<Weather>,
    tileset: Res<Tileset>,
    mut tiles: Query<&mut Sprite, With<DisplayTile>>,
